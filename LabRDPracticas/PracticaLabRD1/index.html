Practica 1

Entradas y Salidas Digitales

En esta practica Conocimos, identificamos y comprobamos el funcionamiento de las entradas y salidas digitales en el módulo Wifi ESP8266. Los materiales que usamos fueron los siguientes:

  • ESP32
  • 2 Botones Push-Button
  • 1 Led Rojo
  • Resistencia 220 Ohms
  • Resistencia 1k Ohms

1er Circuito: Prender 1 Led

Código ESP32:



                      //Prender 1 Led

4to Circuito: Funcion OR

Código Arduino Transmisor:


                    
                    //Funcion OR
                    #define Led 4
                    #define Push1 19
                    #define Push2 23

                    int Button1 = 0;
                    int Button2 = 0;

                    void setup() {
                      Serial.begin(115200);
                      pinMode(Led, OUTPUT);
                      pinMode(Push1, INPUT);
                      pinMode(Push2, INPUT);
                    }

                    void loop() {
                      Button1 = digitalRead(Push1);
                      Button2 = digitalRead(Push2);

                      // Utilizamos la lógica OR para encender el LED si cualquiera de los botones está presionado
                      if (Button1 == 1 || Button2 == 1) {
                        digitalWrite(Led, HIGH);
                        Serial.print("Led Prendido");
                      } else {
                        digitalWrite(Led, LOW);
                        Serial.print("Led Apagado");
                      }
                    }

5to Circuito: Funcion AND

Código ESP32:



                  //Funcion AND
                  #define Led 4
                  #define Push1 19
                  #define Push2 23

                  int Button1 = 0;
                  int Button2 = 0;

                  void setup() {
                    Serial.begin(115200);
                    pinMode(Led, OUTPUT);
                    pinMode(Push1, INPUT);
                    pinMode(Push2, INPUT);
                  }

                  void loop() {
                    Button1 = digitalRead(Push1);
                    Button2 = digitalRead(Push2);

                    // Lógica AND para encender el LED 
                    if (Button1 == 1 && Button2 == 1) {
                      digitalWrite(Led, HIGH);
                      Serial.print("Led Prendido");
                    } else {
                      digitalWrite(Led, LOW);
                      Serial.print("Led Apagado");
                    }
                  }
