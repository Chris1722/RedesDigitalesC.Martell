3er Circuito

Envio Datos PHP ESP32 Local


#include 
#include 

int boton = 33;
int led = 32;

// Variables de estado
int estado_boton = 0;

String url = "https://389e2d6f-f21f-42c6-97bf-791e2f870ea0-00-2kbe8xneipoxk.picard.replit.dev/?";
String sensor = "DATO";

void setup() {
  // Configurar el puerto 32 como salida
  pinMode(led, OUTPUT);

  // Configurar el puerto 33 como entrada
  pinMode(boton, INPUT);
  delay(1000);
  Serial.begin(115200);

  WiFi.begin("Wokwi-GUEST", "");

  while((WiFi.status() != WL_CONNECTED)) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("WiFi Conected");
}

void loop() {
  // Leer el estado del botón
  estado_boton = digitalRead(boton);

  // Si el botón está presionado, encender el LED
  if (estado_boton == 1) {
    digitalWrite(led, HIGH);
  } 
  
  else {
    // Si el botón no está presionado, apagar el LED
    digitalWrite(led, LOW);
  }

  if((WiFi.status() == WL_CONNECTED)) {

    HTTPClient http;
    String message=url+sensor+"="+String(estado_boton);
    http.begin(message);

    int httpCode = http.GET();
    Serial.println(message);
    Serial.println("GET code:");
    Serial.println(httpCode);
    http.end();
  }
  delay(3000);
}
